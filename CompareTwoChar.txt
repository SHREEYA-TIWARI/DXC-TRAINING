public class mfp{
boolean isEqual(char c1,char c2){
   
   String ss=String.valueOf(c1); 
   String sa=String.valueOf(c2); 
    if( ss.equalsIgnoreCase(sa) ) {
            return true;
    }
    else
    return false;
  }
    

    public static void main(String [] args) {
    	char s='A';
    	char s2='a';
    	
    	mfp obj=new mfp();
    	System.out.println(obj.isEqual(s,s2));
    	
    }
}
