public class mfp{
boolean isCapsOnly(String str){
    
    
    char[] charArray = str.toCharArray();
    int cnt=0;
    for(int i=0; i < charArray.length; i++){
        
    if( Character.isLowerCase( charArray[i] )) {
            cnt++;
    }
    }
    if(cnt==0)
    	return true;
    else
    	return false;
    
    }
    

    public static void main(String [] args) {
    	String s="AAAA";
    	mfp obj=new mfp();
    	System.out.println(obj.isCapsOnly(s));
    	
    }
}
