public class mfp{
String hasUpper(String str){
    
    
    char[] charArray = str.toCharArray();
    int cnt=0;
    for(int i=0; i < charArray.length; i++){
        
    if( Character.isUpperCase( charArray[i] ))
            cnt++;
    }
    if(cnt>0) {
    	return "Yes";
    }
    else {
    return "No";
    }
}
    public static void main(String [] args) {
    	String s="hello";
    	mfp obj=new mfp();
    	System.out.println(obj.hasUpper(s));
    	
    }
}
