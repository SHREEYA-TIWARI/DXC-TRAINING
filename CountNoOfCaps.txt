public class mfp{
int countCaps(String data){
    
    
    char[] charArray = data.toCharArray();
    int cnt=0;
    for(int i=0; i < charArray.length; i++){
        
    if( Character.isUpperCase( charArray[i] )) {
            cnt++;
    }
    }
    return cnt;
   
    
    }
    

    public static void main(String [] args) {
    	String s="AAAAaaaaaP";
    	mfp obj=new mfp();
    	System.out.println(obj.countCaps(s));
    	
    }
}
